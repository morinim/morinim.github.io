114  std::uint32_t seed)
115 {

117 
118  const std::uint64_t c1(0x87c37b91114253d5), c2(0x4cf5ad432745937f);
119 
120  // Body.
121  const auto *blocks(reinterpret_cast<const std::uint64_t *>(data));
122  hash_t h(seed, seed);
123 
124  for (std::size_t i(0); i < n_blocks; ++i)
125  {
126  std::uint64_t k1(get_block(blocks, i * 2 + 0));
127  std::uint64_t k2(get_block(blocks, i * 2 + 1));

131  k1 *= c2;

133

135  h.data[0] += h.data[1];
136  h.data[0] = h.data[0] * 5 + 0x52dce729;

146  }

150
