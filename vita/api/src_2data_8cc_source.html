439 std::size_t src_data::load_xrff(const std::string &filename, filter_hook_t ft)
440 {
441  tinyxml2::XMLDocument doc;
442  if (doc.LoadFile(filename.c_str()) != tinyxml2::XML_SUCCESS)
443  return 0;
444 
445  unsigned n_output(0);
446  bool classification(false);
447 
448  // Iterate over `dataset.header.attributes` selection and store all found
449  // attributes in the header vector.
450  tinyxml2::XMLHandle handle(&doc);
451  auto *attributes = handle.FirstChildElement("dataset")
452  .FirstChildElement("header")
453  .FirstChildElement("attributes").ToElement();
454  if (!attributes)
455  return 0;
456 
457  for (auto *attribute = attributes->FirstChildElement("attribute");
458  attribute;
459  attribute = attribute->NextSiblingElement("attribute"))
460  {
461  column a;
462 
463  const char *s = attribute->Attribute("name");
464  if (s)
465  a.name = s;
466 
467  // One can define which attribute should act as output value via the
468  // `class="yes"` attribute in the attribute specification of the header.
469  const bool output(attribute->Attribute("class", "yes"));
470 
471  s = attribute->Attribute("type");
472  std::string xml_type(s ? s : "");
473 
474  s = attribute->Attribute("category");
475  std::string category_name(s ? s : xml_type);
476 
477  if (output)
478  {
479  ++n_output;
480 
481  // We can manage only one output column.
482  if (n_output > 1)
483  return 0;
484 
485  // For classification problems we use discriminant functions, so the
486  // actual output type is always numeric.
487  classification = (xml_type == "nominal" || xml_type == "string");
488  if (classification)
489  {
490  xml_type = "numeric";
491  category_name = "numeric";
492  }
493  }
494 
495  a.category_id = categories_.insert({category_name, from_weka(xml_type),
496  {}});
497 
498  // Store label1... labelN.
499  if (xml_type == "nominal")
500  for (auto *l = attribute->FirstChildElement("label");
501  l;
502  l = l->NextSiblingElement("label"))
503  {
504  const std::string label(l->GetText() ? l->GetText() : "");
505  categories_.add_label(a.category_id, label);
506  }
507 
508  // Output column is always the first one.
509  if (output)
510  header_.insert(header_.begin(), a);
511  else
512  header_.push_back(a);
513  }
514 
515  // XRFF needs informations about the columns.
516  if (!columns())
517  return 0;
518 
519  // If no output column is specified the default XRFF output column is the
520  // last one (and it's the first element of the `header_` vector).
521  if (n_output == 0)
522  {
523  header_.insert(header_.begin(), header_.back());
524  header_.pop_back();
525  }
526 
527  // Category 0 is the output category.
528  swap_category(category_t(0), header_[0].category_id);
529 
530  auto *instances = handle.FirstChildElement("dataset")
531  .FirstChildElement("body")
532  .FirstChildElement("instances").ToElement();
533  if (!instances)
534  return 0;
535 
536  for (auto *i = instances->FirstChildElement("instance");
537  i;
538  i = i->NextSiblingElement("instance"))
539  {
540  std::vector<std::string> record;
541  for (auto *v = i->FirstChildElement("value");
542  v;
543  v = v->NextSiblingElement("value"))
544  record.push_back(v->GetText() ? v->GetText() : "");
545 
546  if (ft && ft(record) == false)
547  continue;
548 
549  const auto instance(to_example(record, classification, false));
550 
551  if (instance.input.size() + 1 == columns())
552  push_back(instance);
553  else
554  print.warning("Malformed example ", size(), " skipped");
555  }
556 
557  return debug() ? size() : static_cast<std::size_t>(0);
558 }

606 std::size_t src_data::load_csv(const std::string &filename, filter_hook_t ft)
607 {
608  std::ifstream from(filename);
609  if (!from)
610  return 0;
611 
612  bool classification(false), format(columns());
613 
614  for (auto record : csv_parser(from).filter_hook(ft))
615  {
616  const auto fields(record.size());
617 
618  // If we don't know the dataset format, the first line will be used to
619  // learn it.
620  if (!format)
621  {
622  classification = !is_number(record[0]);
623 
624  header_.reserve(fields);
625 
626  for (auto field(decltype(fields){0}); field < fields; ++field)
627  {
628  assert(!size()); // If we have data then data format must be known
629 
630  std::string s_domain(is_number(record[field])
631  ? "numeric"
632  : "string" + std::to_string(field));
633 
634  // For classification problems we use discriminant functions, so the
635  // actual output type is always numeric.
636  if (field == 0 && classification)
637  s_domain = "numeric";
638 
639  const domain_t domain(s_domain == "numeric" ? domain_t::d_double
640  : domain_t::d_string);
641 
642  const category_t tag(categories_.insert({s_domain, domain, {}}));
643 
644  header_.push_back({"", tag});
645  }
646 
647  format = true;
648  } // if (!format)
649 
650  if (fields != columns()) // skip lines with wrong number of columns
651  continue;
652 
653  const auto instance(to_example(record, classification, true));
654 
655  if (instance.input.size() + 1 == columns())
656  push_back(instance);
657  else
658  print.warning("Malformed line ", size(), " skipped");
659  }
660 
661  return debug() ? size() : static_cast<std::size_t>(0);
662 }

741 domain_t src_data::from_weka(const std::string &n)
742 {
743  static const std::map<const std::string, domain_t> map(
744  {
745  // This type is vita-specific (not standard).
746  {"boolean", domain_t::d_bool},
747 
748  {"integer", domain_t::d_int},
749 
750  // Real and numeric are treated as double precision number (d_double).
751  {"numeric", domain_t::d_double},
752  {"real", domain_t::d_double},
753 
754  // Nominal values are defined by providing a list of possible values.
755  {"nominal", domain_t::d_string},
756 
757  // String attributes allow us to create attributes containing arbitrary
758  // textual values. This is very useful in text-mining applications.
759  {"string", domain_t::d_string}
760 
761  // {"date", ?}, {"relational", ?}
762  });
763 
764  const auto &i(map.find(n));
765  return i == map.end() ? domain_t::d_void : i->second;
