474 const symbol &symbol_set::collection::sum_container::roulette() const
475 {
476  Expects(sum());
477 
478  const auto slot(random::sup(sum()));
479 
480  std::size_t i(0);
481  for (auto wedge(elems_[i].weight);
482  wedge <= slot;
483  wedge += elems_[++i].weight)
484  {}
485 
486  assert(i < elems_.size());
487  return *elems_[i].sym;
488 
489  // The so called roulette-wheel selection via stochastic acceptance:
490  //
491  // for (;;)
492  // {
493  // const symbol *s(random::element(elems));
494  //
495  // if (random::sup(max) < s->weight)
496  // return *s;
497  // }
498  //
499  // Internal tests have proved this is slower for Vita.
500 
501  // This is a different approach from Eli Bendersky
502  // (<http://eli.thegreenplace.net/>):
503  //
504  // unsigned total(0);
505  // std::size_t winner(0);
506  //
507  // for (std::size_t i(0); i < size(); ++i)
508  // {
509  // total += elems[i].weight;
510  // if (random::sup(total + 1) < elems[i].weight)
511  // winner = i;
512  // }
513  // return *elems[winner].sym;
514  //
515  // The interesting property of this algorithm is that you don't need to
516  // know the sum of weights in advance in order to use it. The method is
517  // cool, but slower than the standard roulette.
518 }
