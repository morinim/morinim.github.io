492 const symbol &symbol_set::collection::sum_container::roulette() const
493 {
494  Expects(sum());
495 
496  const auto slot(random::sup(sum()));
497 
498  std::size_t i(0);
499  for (auto wedge(elems_[i].weight);
500  wedge <= slot;
501  wedge += elems_[++i].weight)
502  {}
503 
504  assert(i < elems_.size());
505  return *elems_[i].sym;
506 
507  // The so called roulette-wheel selection via stochastic acceptance:
508  //
509  // for (;;)
510  // {
511  // const symbol *s(random::element(elems));
512  //
513  // if (random::sup(max) < s->weight)
514  // return *s;
515  // }
516  //
517  // Internal tests have proved this is slower for Vita.
518 
519  // This is a different approach from Eli Bendersky
520  // (<http://eli.thegreenplace.net/>):
521  //
522  // weight_t total(0);
523  // std::size_t winner(0);
524  //
525  // for (std::size_t i(0); i < size(); ++i)
526  // {
527  // total += elems[i].weight;
528  // if (random::sup(total + 1) < elems[i].weight)
529  // winner = i;
530  // }
531  // return *elems[winner].sym;
532  //
533  // The interesting property of this algorithm is that you don't need to
534  // know the sum of weights in advance in order to use it. The method is
535  // cool, but slower than the standard roulette.
536 }
