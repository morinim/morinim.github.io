366 void i_mep::pack(const locus &l, std::vector<unsigned char> *const p) const
367 {
368  const gene &g(genome_(l));
369 
370  // Although 16 bit are enough to contain opcodes and parameters, they are
371  // usually stored in unsigned variables (i.e. 32 or 64 bit) for performance
372  // reasons.
373  // Anyway before hashing opcodes/parameters we convert them to 16 bit types
374  // to avoid hashing more than necessary.
375  const auto opcode(static_cast<std::uint16_t>(g.sym->opcode()));
376  assert(g.sym->opcode() <= std::numeric_limits<decltype(opcode)>::max());
377 
378  // DO NOT CHANGE reinterpret_cast type to std::uint8_t since even if
379  // std::uint8_t has the exact same size and representation as
380  // unsigned char, if the implementation made it a distinct, non-character
381  // type, the aliasing rules would not apply to it
382  // (see <http://stackoverflow.com/q/16138237/3235496>)
383  auto s1 = reinterpret_cast<const unsigned char *>(&opcode);
384  for (std::size_t i(0); i < sizeof(opcode); ++i)
385  p->push_back(s1[i]);
386 
387  auto arity(g.sym->arity());
388  if (arity)
389  {
390  decltype(arity) i(0);
391  do
392  pack(g.arg_locus(i), p);
393  while (++i < arity);
394  }
395  else
396  {
397  if (terminal::cast(g.sym)->parametric())
398  {
399  const auto param(g.par);
400 
401  auto s2 = reinterpret_cast<const unsigned char *>(&param);
402  for (std::size_t i(0); i < sizeof(param); ++i)
403  p->push_back(s2[i]);
404  }
405  }

572 bool i_mep::load_impl(std::istream &in, const problem &p)
573 {
574  unsigned rows, cols;
575  if (!(in >> rows >> cols))
576  return false;
577 
578  // The matrix class has a basic support for serialization but we cannot
579  // take advantage of it here: the gene class needs a special management
580  // (among other things it needs access to the symbol_set to decode the
581  // symbols).
582  decltype(genome_) genome(rows, cols);
583  for (auto &g : genome)
584  {
585  opcode_t opcode;
586  if (!(in >> opcode))
587  return false;
588 
589  gene temp;
590 
591  temp.sym = p.sset.decode(opcode);
592  if (!temp.sym)
593  return false;
594 
595  if (temp.sym->terminal() && terminal::cast(temp.sym)->parametric())
596  if (!(in >> temp.par))
597  return false;
598 
599  auto arity(temp.sym->arity());
600  if (arity)
601  {
602  temp.args.resize(arity);
603 
604  for (auto &arg : temp.args)
605  if (!(in >> arg))
606  return false;
607  }
608 
609  g = temp;
610  }
611 
612  auto best(locus::npos());
613 
614  if (rows && !(in >> best.index >> best.category))
615  return false;
616 
617  best_ = best;
618  genome_ = genome;
619 
620  return true;
621 }

664 {
665  i_mep ret(*this);
666 
667  // std::map needs a compare function and there isn't a predefined less
668  // operator for gene class.
669  struct cmp_gene
670  {
671  bool operator()(const gene &a, const gene &b) const
672  {
673  if (a.sym->opcode() < b.sym->opcode())
674  return true;
675 
676  if (a.sym->opcode() == b.sym->opcode())
677  {
678  if (a.sym->terminal())
679  return terminal::cast(a.sym)->parametric() ? a.par < b.par : false;
680 
681  auto arity(a.sym->arity());
682  for (decltype(arity) i(0); i < arity; ++i)
683  if (a.args[i] < b.args[i])
684  return true;
685  }
686 
687  return false;
688  }
689  };
690 
691  std::map<gene, locus, cmp_gene> new_locus;
692 
693  for (index_t i(size()); i > 0; --i)
694  for (category_t c(0); c < genome_.cols(); ++c)
695  {
696  gene &g(ret.genome_(i - 1, c));
697 
698  const auto arity(g.sym->arity());
699  for (auto p(decltype(arity){0}); p < arity; ++p)
700  {
701  const auto where(new_locus.find(ret[g.arg_locus(p)]));
702  if (where != new_locus.end())
703  {
704  assert(where->second.index <=
705  std::numeric_limits<gene::packed_index_t>::max());
706 
707  g.args[p] = static_cast<gene::packed_index_t>(where->second.index);
708  }
709  }
710 
711  new_locus.insert({g, {i - 1, c}});
712  }
713 
714  return ret;

788 i_mep crossover(const i_mep &lhs, const i_mep &rhs)
789 {
790  Expects(lhs.debug());
791  Expects(rhs.debug());
792  Expects(lhs.size() == rhs.size());
793 
794  const bool b(random::boolean());
795  const i_mep &from(b ? rhs : lhs);
796  i_mep to(b ? lhs : rhs);
797 
798  switch (from.active_crossover_type_)
799  {
800  case i_mep::crossover_t::one_point:
801  {
802  const auto i_sup(from.size());
803  const auto c_sup(from.categories());
804  const auto cut(random::between<index_t>(1, i_sup - 1));
805 
806  for (index_t i(cut); i < i_sup; ++i)
807  for (category_t c(0); c < c_sup; ++c)
808  {
809  const locus l{i, c};
810  to.genome_(l) = from[l];
811  }
812  }
813  break;
814 
815  case i_mep::crossover_t::two_points:
816  {
817  const auto i_sup(from.size());
818  const auto c_sup(from.categories());
819 
820  const auto cut1(random::sup(i_sup - 1));
821  const auto cut2(random::between(cut1 + 1, i_sup));
822 
823  for (index_t i(cut1); i != cut2; ++i)
824  for (category_t c(0); c < c_sup; ++c)
825  {
826  const locus l{i, c};
827  to.genome_(l) = from[l];
828  }
829  }
830  break;
831 
832  case i_mep::crossover_t::uniform:
833  {
834  const auto i_sup(from.size());
835  const auto c_sup(from.categories());
836 
837  for (index_t i(0); i != i_sup; ++i)
838  for (category_t c(0); c < c_sup; ++c)
839  if (random::boolean())
840  {
841  const locus l{i, c};
842  to.genome_(l) = from[l];
843  }
844  }
845  break;
846 
847  default: // Tree crossover
848  {
849  auto crossover_ = [&](locus l, const auto &lambda) -> void
850  {
851  to.genome_(l) = from[l];
852 
853  if (!from[l].sym->terminal())
854  {
855  const auto &f(*function::cast(from[l].sym));
856  for (unsigned i(0); i < f.arity(); ++i)
857  lambda(from[l].arg_locus(i), lambda);
858  }
859  };
860 
861  const auto delta(random::sup(from.active_symbols()));
862  crossover_(std::next(from.begin(), delta).locus(), crossover_);
863  }
864  break;
865  }
866 
867  to.active_crossover_type_ = from.active_crossover_type_;
868  to.set_older_age(from.age());
869  to.signature_.clear();
870 
871  Ensures(to.debug());
872  return to;
873 }
