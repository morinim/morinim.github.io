35  std::size_t size)

236 std::size_t src_problem::load_symbols(const std::string &s_file)
237 {
238  setup_terminals_from_data();
239 
240  // Prints the list of categories as inferred from the dataset.
241  for (const category &c : dat_.categories())
242  print.debug("Using ", c);
243 
244  std::set<category_t> used_categories;
245  for (category_t i(0); i < categories(); ++i)
246  used_categories.insert(used_categories.end(), i);
247 
248  tinyxml2::XMLDocument doc;
249  if (doc.LoadFile(s_file.c_str()) != tinyxml2::XML_SUCCESS)
250  return 0;
251 
252  std::size_t parsed(0);
253 
254  // When I wrote this, only God and I understood what I was doing.
255  // Now, God only knows.
256  tinyxml2::XMLHandle handle(&doc);
257  auto *symbolset(handle.FirstChildElement("symbolset").ToElement());
258 
259  if (!symbolset)
260  return 0;
261 
262  for (auto *s(symbolset->FirstChildElement("symbol"));
263  s;
264  s = s->NextSiblingElement("symbol"))
265  {
266  const char *c_sym_name(s->Attribute("name"));
267  if (!c_sym_name)
268  {
269  print.error("Skipped unnamed symbol in symbolset");
270  continue;
271  }
272  const std::string sym_name(c_sym_name);
273 
274  const char *sym_sig(s->Attribute("signature"));
275 
276  if (sym_sig) // single category, uniform initialization
277  {
278  for (auto tag : used_categories)
279  if (compatible({tag}, {std::string(sym_sig)}))
280  {
281  const auto n_args(factory_.args(sym_name));
282  std::string signature(sym_name + ":");
283 
284  for (auto j(decltype(n_args){0}); j < n_args; ++j)
285  signature += " " + dat_.categories().find(tag).name;
286  print.debug("Adding to symbol set ", signature);
287 
288  env.sset->insert(factory_.make(sym_name, cvect(n_args, tag)));
289  }
290  }
291  else // !sym_sig => complex signature
292  {
293  auto *sig(s->FirstChildElement("signature"));
294  if (!sig)
295  {
296  print.error("Skipping " + sym_name + " symbol (empty signature)");
297  continue;
298  }
299 
300  std::vector<std::string> args;
301  for (auto *arg(sig->FirstChildElement("arg"));
302  arg;
303  arg = arg->NextSiblingElement("arg"))
304  {
305  if (!arg->GetText())
306  {
307  print.error("Skipping " + sym_name + " symbol (wrong signature)");
308  args.clear();
309  break;
310  }
311 
312  args.push_back(arg->GetText());
313  }
314 
315  // From the list of all the sequences with repetition of `args.size()`
316  // elements...
317  const auto sequences(detail::seq_with_rep(used_categories, args.size()));
318 
319  // ...we choose those compatible with the xml signature of the current
320  // symbol.
321  for (const auto &seq : sequences)
322  if (compatible(seq, args))
323  {
324  std::string signature(sym_name + ":");
325  for (const auto &j : seq)
326  signature += " " + dat_.categories().find(j).name;
327  print.debug("Adding to symbol set ", signature);
328 
329  env.sset->insert(factory_.make(sym_name, seq));
330  }
331  }
332 
333  ++parsed;
334  }

336  return parsed;
